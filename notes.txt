docker run -ti [image] [comando]
	avvia l'immagine
	-ti = tastiera + interattivo
	--rm uccide lo zombie alla fine
	-v[source]:[destination] mappo directory fuori container dentro il contaiener
		EX: /dev per leggere le periferche come telecamera
	
docker ps
	container attivi
	-a mostra anche gli zombie
	
docker run [container zombie]
	risveglia zombie
	
docker attach [container]
	rientra nel container
	
docker stop [container]
	fa diventare container zombie e sospende processi
	
docker rm [container]
	
docker rmi [image]

docker commit [container]
	trasforma container in immagine
